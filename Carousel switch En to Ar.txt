  <script>
    // Function to init carousel
    function initCarousel(isRTL) {
      $(".owl-carousel").owlCarousel("destroy"); // destroy existing
      $(".owl-carousel").owlCarousel({
        loop: false,
        margin: 15,
        nav: true,
        dots: true,
        autoplay: false,
        rtl: isRTL, // switch direction
        responsive: {
          0: { items: 2 },
          576: { items: 3 },
          992: { items: 4 },
          1200: { items: 5 }
        }
      });
    }

    $(document).ready(function () {
      // Detect initial language
      const isRTL = $("html").attr("lang") === "ar";
      initCarousel(isRTL);

      // Switch to English
      $("#to-en").click(function () {
        $("html").attr("lang", "en");
        $("body").css("direction", "ltr");
        initCarousel(false);
      });

      // Switch to Arabic
      $("#to-ar").click(function () {
        $("html").attr("lang", "ar");
        $("body").css("direction", "rtl");
        initCarousel(true);
      });
    });
  </script>
